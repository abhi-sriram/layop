class ImageUpload extends StatefulWidget {
  const ImageUpload({Key? key}) : super(key: key);

  @override
  _ImageUploadState createState() => _ImageUploadState();
}

class _ImageUploadState extends State<ImageUpload> {
  File? _image;
  final service = Service();
  PickedFile? _imageFIle;

  Future obticam(ImageSource source) async {
    final _image = await ImagePicker().pickImage(source: source);
    if (_image == null) return;
    final tempim = File(_image.path);
    setState(() {
      this._image = tempim;
    });
  }

  @override
  void initState() {
    super.initState();
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: Text("mettre a jour image"),
        centerTitle: true,
      ),
      body: Column(
          children: [
        Container(
          width: MediaQuery.of(context).size.width,
          height: 200.0,
          child: Center(
            child: _image == null ? Text("Aucune image") : Image.file(_image!),
          ),
        ),
        Row(
          mainAxisAlignment: MainAxisAlignment.spaceEvenly,
          children: [
            ElevatedButton(
                onPressed: () {
                  obticam(ImageSource.camera);
                },
                child: Text("Camera")),
            ElevatedButton(
                onPressed: () {
                  obticam(ImageSource.gallery);
                },
                child: Text("gallerie")),
          ],
        ),
            InkWell(onTap: () async{

}
